#include <ESP8266WiFi.h>
#include <PubSubClient.h>

int sense_Pin = 0; // sensor input at Analog pin A0
int value = 0;
const char* ssid = "home114";                   // wifi ssid
const char* password =  "openhabiann";         // wifi password
const char* mqttServer = "192.168.0.8";    // IP adress Raspberry Pi
const int mqttPort = 1883;
const char* mqttUser = "rsraju";      // if you don't have MQTT Username, no need input
const char* mqttPassword = "18mcpc13";  // if you don't have MQTT Password, no need input

WiFiClient espClient;
PubSubClient client(espClient);

void setup(){

  Serial.begin(115200);
  
  WiFi.begin(ssid, password);
  Serial.println("Checkpoint 1");
  while(WiFi.status() != WL_CONNECTED) 
  {
    delay(500);
    Serial.println("Connecting to WiFi..");
  }
  Serial.println("Connected to the WiFi network");

  client.setServer(mqttServer, mqttPort);
  client.setCallback(callback);
  Serial.println("Checkpoint 2");
  while (!client.connected()) {
    Serial.println("Connecting to MQTT...");

    if (client.connect("ESP8266Client")) {

      Serial.println("connected");

    } else {

      Serial.print("failed with state ");
      Serial.print(client.state());
      delay(2000);

    }
  }

  //client.publish("esp8266", "Hello Raspberry Pi");
  //client.subscribe("esp8266");

}

void callback(char* topic, byte* payload, unsigned int length) {

  Serial.print("Message arrived in topic: ");
  Serial.println(
    topic);

  Serial.print("Message:");
  for (int i = 0; i < length; i++) {
    Serial.print((char)payload[i]);
  }

  Serial.println();
  Serial.println("-----------------------");

}

void loop() {

    value= analogRead(sense_Pin);
    value= value/10;
    char s[16];
    
    client.subscribe("esp8266");
    client.publish("esp8266", itoa(value, s, 10));
    delay(1000);
    client.loop();
}
